2.3 Proximity to Day 28 (Housing Benefit Payment)

Code 
# Function to calculate distance to a fixed payment day
calc_payment_proximity <- function(survey_day, payment_day, days_in_month = 30) {
  if (survey_day >= payment_day) {
    days_since <- survey_day - payment_day
    days_until <- (days_in_month - survey_day) + payment_day
  } else {
    days_since <- (days_in_month - payment_day) + survey_day
    days_until <- payment_day - survey_day
  }
  circular <- min(days_since, days_until)
  return(list(days_since = days_since, days_until = days_until, circular = circular))
}

# Vectorized version
calc_proximity_vec <- function(survey_days, payment_day, days_in_month = 30) {
  n <- length(survey_days)
  days_since <- numeric(n)
  days_until <- numeric(n)

  for (i in seq_len(n)) {
    if (is.na(survey_days[i])) {
      days_since[i] <- NA
      days_until[i] <- NA
    } else {
      result <- calc_payment_proximity(survey_days[i], payment_day, days_in_month)
      days_since[i] <- result$days_since
      days_until[i] <- result$days_until
    }
  }
  circular <- pmin(days_since, days_until)
  return(data.frame(days_since = days_since, days_until = days_until, circular = circular))
}

# Calculate proximity to 28th
prox_28 <- calc_proximity_vec(df$day_of_month, payment_day = 28)
df$prox_28_since <- prox_28$days_since
df$prox_28_until <- prox_28$days_until
df$prox_28_circ <- prox_28$circular

2.4 Housing Benefit Status

Panel propagation logic: if someone reports receiving a benefit in ANY wave, code them as beneficiary in ALL waves, UNLESS they select “None of these programs” in that specific wave.

Code 
# Raw indicator
df$raw_mprog_housingbnft <- ifelse(!is.na(df$mprog_housingbnft), 1, 0)

# "None of these programs" indicator
df$selected_none <- ifelse(!is.na(df$mprog_none), 1, 0)

# Get IDs who ever reported housing benefit
ever_housing <- df %>%
  filter(raw_mprog_housingbnft == 1) %>%
  pull(ID) %>%
  unique()

# Create beneficiary variable with propagation
df <- df %>%
  mutate(
    ben_housingbnft = case_when(
      selected_none == 1 ~ 0,
      ID %in% ever_housing ~ 1,
      TRUE ~ 0
    )
  )
